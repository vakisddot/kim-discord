CREATE TABLE descriptions (serverId INTEGER NOT NULL, description VARCHAR(255) NOT NULL)
CREATE TABLE phrases (serverId INTEGER NOT NULL, phrase VARCHAR(95) NOT NULL)
CREATE TABLE "servers" (
	"id"	INTEGER NOT NULL,
	"name"	VARCHAR(95),
	"spoilerMode"	BOOL DEFAULT 1,
	"canKick"	TINYINT DEFAULT 0,
	"swearThreshold"	TINYINT DEFAULT 0,
	"replyInDm"	TINYINT DEFAULT 1
)
CREATE TABLE users (id INTEGER NOT NULL, serverId INTEGER NOT NULL, swearCount TINYINT DEFAULT 0)
CREATE TABLE words (serverId INTEGER NOT NULL, "word" VARCHAR(95) NOT NULL)
